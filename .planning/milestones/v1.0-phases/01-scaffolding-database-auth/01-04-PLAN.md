```typescript
export const brands = sqliteTable('brands', {
  id: integer().primaryKey({ autoIncrement: true }),
  name: text().notNull(),
  niche: text().notNull(),
  voiceTone: text('voice_tone').notNull(),
  targetAudience: text('target_audience'),
  goals: text(),
  topics: text({ mode: 'json' }).$type<string[]>(),
  dosList: text('dos_list', { mode: 'json' }).$type<string[]>(),
  dontsList: text('donts_list', { mode: 'json' }).$type<string[]>(),
  examplePosts: text('example_posts', { mode: 'json' }).$type<string[]>(),
  platformNotes: text('platform_notes', { mode: 'json' }).$type<Record<string, string>>(),
  ctaText: text('cta_text'),
  bioTemplate: text('bio_template'),
  bioLink: text('bio_link'),
  bannedHashtags: text('banned_hashtags', { mode: 'json' }).$type<string[]>(),
  primaryColor: text('primary_color'),
  secondaryColor: text('secondary_color'),
  logoUrl: text('logo_url'),
  watermarkPosition: text('watermark_position', { enum: ['top-left','top-right','bottom-left','bottom-right'] }),
  watermarkOpacity: integer('watermark_opacity'),
  warmupDate: text('warmup_date'),
  createdAt: text('created_at').notNull().$defaultFn(() => new Date().toISOString()),
  updatedAt: text('updated_at').notNull().$defaultFn(() => new Date().toISOString()),
})

export const socialAccounts = sqliteTable('social_accounts', {
  id: integer().primaryKey({ autoIncrement: true }),
  brandId: integer('brand_id').notNull().references(() => brands.id),
  platform: text().notNull(),
  username: text().notNull(),
  status: text({ enum: ['connected', 'disconnected'] }).notNull().default('connected'),
  failureCount: integer('failure_count').notNull().default(0),
  uploadPostUsername: text('upload_post_username'),
  createdAt: text('created_at').notNull().$defaultFn(() => new Date().toISOString()),
})
```

    **Create `src/app/actions/brands.ts`.**
    - Add 'use server' directive at top
    - Import getDb from '@/db', brands schema, eq from drizzle-orm, revalidatePath from 'next/cache', redirect from 'next/navigation'

    **`createBrand(formData: FormData): Promise<void>`**
    - Extract required fields: name, niche, voiceTone (throw if missing)
    - Extract optional string fields: targetAudience, goals, ctaText, bioTemplate, bioLink, logoUrl, warmupDate, primaryColor, secondaryColor, watermarkPosition
    - Extract watermarkOpacity as integer (parse from string)
    - Extract JSON array fields from comma-separated or newline-separated input: topics, dosList, dontsList, examplePosts, bannedHashtags — split by newline, filter empty, trim each
    - Extract platformNotes as JSON object (keys: twitter, instagram, linkedin, tiktok — each from separate form field)
    - Insert into brands table with getDb()
    - revalidatePath('/brands')
    - redirect('/brands')

    **`updateBrand(id: number, formData: FormData): Promise<void>`**
    - Same field extraction as createBrand
    - Update brands table where id matches
    - Set updatedAt to new Date().toISOString()
    - revalidatePath(`/brands/${id}`)
    - revalidatePath('/brands')
    - redirect(`/brands/${id}`)

    **`deleteBrand(id: number, confirmName: string): Promise<{ error?: string }>`**
    - Query brand by id, verify confirmName matches brand.name exactly
    - If mismatch: return { error: 'Brand name does not match' }
    - Delete all socialAccounts where brandId = id (cascade)
    - Delete brand
    - revalidatePath('/brands')
    - redirect('/brands')

       - Example Posts (Textarea, one per line, optional)
