```typescript
export const brands = sqliteTable('brands', {
  id: integer().primaryKey({ autoIncrement: true }),
  name: text().notNull(),
  niche: text().notNull(),
  primaryColor: text('primary_color'),
  // ...
})

export const posts = sqliteTable('posts', {
  id: integer().primaryKey({ autoIncrement: true }),
  brandId: integer('brand_id').notNull().references(() => brands.id),
  status: text({ enum: ['draft', 'scheduled', 'published', 'failed'] }),
  scheduledAt: text('scheduled_at'),
  publishedAt: text('published_at'),
  qualityScore: integer('quality_score'),
  createdAt: text('created_at'),
  // ...
})

export const postPlatforms = sqliteTable('post_platforms', {
  postId: integer('post_id').notNull().references(() => posts.id),
  platform: text().notNull(),
  status: text({ enum: ['pending', 'published', 'failed'] }),
  // ...
})

export const postAnalytics = sqliteTable('post_analytics', {
  id: integer().primaryKey({ autoIncrement: true }),
  postId: integer('post_id').notNull().references(() => posts.id),
  platform: text().notNull(),
  views: integer(),
  likes: integer(),
  comments: integer(),
  shares: integer(),
  engagementScore: integer('engagement_score'),
  performerTier: text('performer_tier', { enum: ['top', 'average', 'under'] }),
  collectedAt: text('collected_at').notNull(),
  createdAt: text('created_at'),
})

export const activityLog = sqliteTable('activity_log', {
  id: integer().primaryKey({ autoIncrement: true }),
  brandId: integer('brand_id').references(() => brands.id),
  type: text().notNull(),
  level: text({ enum: ['info', 'warn', 'error'] }),
  message: text().notNull(),
  metadata: text({ mode: 'json' }),
  createdAt: text('created_at'),
})
```
